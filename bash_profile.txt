export GOPATH=$HOME/Documents/Git/GoProjects
export PATH=$PATH:$GOPATH/bin
export CLICOLOR=1
export LSCOLORS=gxFxCxDxBxexexaxaxaxgx


# Git branch in prompt.
parse_git_branch() {
    git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/ (\1)/'
}

export PS1="\n\033[0;32m\]\w\\033[0;37m\]\$(parse_git_branch)\[\n\033[1;37m\]> \033[00m\]"
